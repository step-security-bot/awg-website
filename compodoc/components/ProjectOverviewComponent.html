import { AfterViewInit, Component, OnDestroy, TemplateRef, ViewChild } from '@angular/core';

import { PortalService } from '@awg-core/page/page-services/portal.service';

@Component({
    selector: 'awg-project-overview',
    templateUrl: './project-overview.component.html',
    styleUrls: ['./project-overview.component.css'],
})
export class ProjectOverviewComponent implements AfterViewInit, OnDestroy {
    @ViewChild('rightPanelPortal') rightPanelPortal: TemplateRef<unknown>;

    constructor(private _portalService: PortalService) {}

    ngAfterViewInit() {
        this._portalService.updateRightPanelPortalData(this.rightPanelPortal);
    }

    ngOnDestroy() {
        this._portalService.clearRightPanelPortalData();
    }
}

<p class="para" i18n>
    Die Anton Webern Gesamtausgabe (AWG) ist eine historisch-kritische Edition, die das gesamte kompositorische Schaffen
    Anton Weberns der Öffentlichkeit in wissenschaftlich angemessener und der musikalischen Praxis dienender Form
    zugänglich machen will. Die Edition beinhaltet nicht nur die von Webern selbst zum Druck beförderten Werke, sondern
    auch deren unpublizierte Fassungen, zu Lebzeiten unveröffentlichte Kompositionen, Jugend- und Studienkompositionen
    sowie Fragmente, Skizzen und Bearbeitungen.
</p>

<p class="para" i18n>
    Die AWG entsteht am
    <a href="http://mws.unibas.ch/">Musikwissenschaftlichen Seminar der Universität Basel</a> in enger inhaltlicher
    Zusammenarbeit mit der <a href="http://www.paul-sacher-stiftung.ch/de/home.html">Paul Sacher Stiftung, Basel</a>, wo
    der größte Teil der überlieferten Webern-Quellen aufbewahrt wird. Seit Oktober 2006 wird die Gesamtausgabe vom
    <a href="http://p3.snf.ch/project-143565">Schweizerischen Nationalfonds</a> und seit 2019 von der
    <a href="https://sagw.ch/sagw/">Schweizerischen Akademie der Geistes- und Sozialwissenschaften</a>
    gefördert. Außerdem wurde in einer Kooperation mit der
    <a href="https://www.oeaw.ac.at/">Österreichischen Akademie der Wissenschaften</a> ab 2008 eine
    <a
        href="https://www.oeaw.ac.at/acdh/musikwissenschaft/forschung/musikedition-und-quellendokumentation/anton-webern/"
        >Wiener Arbeitsstelle</a
    >
    eingerichtet, die Dokumente zur Werkgeschichte, Rezeption und zum historischen Kontext recherchiert. Diese
    Zusammenarbeit wurde durch Fördermittel der
    <a
        href="http://www.evs-musikstiftung.ch/de/preise/preise/foerderprojekte/symposienpublikationen/anton-webern-briefwechsel-mit-der-universal-0"
        >Ernst von Siemens Musikstiftung</a
    >ausgebaut.
</p>

<p class="para" i18n>
    Erscheinen wird die AWG in einer Print-/Online-Hybrid-<a [routerLink]="['/edition', 'overview']">Edition</a>, deren
    gedruckte Bände bei der <a href="http://www.universaledition.com/">Universal Edition, Wien,</a> publiziert und deren
    online zugängliche Bestandteile und Datenbestände als Open Data über eine eigene Webapplikation (<a
        href="https://edition.anton-webern.ch"
        >https://edition.anton-webern.ch</a
    >) präsentiert und über das Software-Framework
    <a href="https://www.dasch.swiss/dsp-app">DaSCH Service Platform (DSP)</a>/<a href="https://www.salsah.org/"
        >SALSAH</a
    >
    des <a href="http://www.dhlab.unibas.ch/">Digital Humanities Lab</a> der Universität Basel aufbereitet werden. Zudem
    werden die Datenbestände durch das
    <a href="https://dasch.swiss/">Swiss National Data & Service Center for the Humanities (DaSCH)</a> zugänglich
    gehalten und langfristig aufbewahrt.
</p>

<p class="para" i18n>
    Die AWG versteht sich als Kompetenzzentrum, in dem neben den editorischen auch archivarische, dokumentarische,
    analytisch-forschende, kommunizierende, dienstleistende und organisatorische Aktivitäten vernetzt werden.
</p>
