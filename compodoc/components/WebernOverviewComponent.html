import { AfterViewInit, Component, OnDestroy, TemplateRef, ViewChild } from '@angular/core';

import { PortalService } from '@awg-core/page/page-services/portal.service';

@Component({
    selector: 'awg-webern-overview',
    templateUrl: './webern-overview.component.html',
    styleUrls: ['./webern-overview.component.css'],
})
export class WebernOverviewComponent implements AfterViewInit, OnDestroy {
    @ViewChild('rightPanelPortal') rightPanelPortal: TemplateRef<unknown>;

    constructor(private _portalService: PortalService) {}

    ngAfterViewInit() {
        this._portalService.updateRightPanelPortalData(this.rightPanelPortal);
    }

    ngOnDestroy() {
        this._portalService.clearRightPanelPortalData();
    }
}

<p i18n>
    Die Anton Webern Gesamtausgabe erarbeitet im Zuge der Edition sämtlicher Kompositionen Anton Weberns eine möglichst
    lückenlose Dokumentation einerseits der Biographie des Komponisten, andererseits der Entstehungs-, Publikations- und
    Aufführungsgeschichte seiner Werke. Zu diesem Zweck werden auf der vom Digital Humanities Lab (DHLab) der
    philosophisch-historischen Fakultät der Universität Basel entwickelten Forschungsplattform DSP
    (<strong>D</strong>aSCH <strong>S</strong>ervice <strong>P</strong>latform; Zugang: SALSAH) Informationen
    unterschiedlicher Art – sowohl materielle Objekte als auch immaterielle Ereignisse – archiviert und verknüpft: Eine
    <a [routerLink]="['/webern', 'chronology']">Chronologie</a>, die Weberns Leben privat wie auch beruflich als Ganzes
    wie auch im Detail beschreibt und die Entwicklung seiner Werke dokumentiert, wird durch Verweise auf materielle
    Zeugnisse (z. B. Korrespondenzen, wie sie in der Rubrik
    <a [routerLink]="['/webern', 'letters']">Briefe</a> archiviert und kategorisiert werden) verifiziert.
    Schlagworthafte Kurzinformationen zu relevanten <a [routerLink]="['/webern', 'persons']">Personen</a> dienen der
    besseren Übersicht über Kontakte und das soziale Umfeld des Komponisten.
</p>

<h4 class="para" i18n>Kurz-Biographie</h4>

<p i18n>
    Am 3. Dezember 1883 in Wien geboren, wuchs Anton Webern in Wien, Graz und Klagenfurt sowie auf dem väterlichen Gut
    Preglhof in Kärnten auf. In Klagenfurt, wo er auch das humanistische Gymnasium besuchte, erhielt er seine erste
    musikalische und musiktheoretische Ausbildung und sammelte praktische Erfahrung als Cellist im örtlichen
    Konzertvereinsorchester. Ab dem Herbstsemester 1902 studierte Webern Musikwissenschaft im von Guido Adler geleiteten
    musikwissenschaftlichen Institut in Wien, wo er 1906 mit einer Dissertation zu Heinrich Isaacs
    <em>Choralis Constantinus</em> abschloss. 1904–1908 war er Kompositionsschüler Arnold Schönbergs, dem er zeitlebens
    eng verbunden blieb. Versuche einer Karriere als fest engagierter Dirigent an diversen Musiktheatern (Teplitz, Bad
    Ischl, Danzig, Stettin, Prag) scheiterten.
</p>

<p i18n>
    Nach dem Ersten Weltkrieg unterrichtete Webern Privatschüler in Musiktheorie und Komposition und arbeitete u. a. als
    Vortragsmeister beim Verein für musikalische Privataufführungen (1918–1922), als Klavierlehrer und Chorleiter am
    Israelitischen Blindeninstitut in Wien (1925–1931) oder übernahm (in seinen letzten Lebensjahren) auch Aufgaben als
    Arrangeur fremder Werke für die Universal Edition.
</p>

<p i18n>
    Als Dirigent verschiedenster Chöre und Orchester (z. B. des Wiener Schubertbundes, des Mödlinger
    Männergesangvereins, der Arbeiter-Symphoniekonzerte der Sozialdemokratischen Kunststelle in Wien) wurde er in den
    Zwanziger- und Dreissigerjahren einem breiteren Publikum bekannt. Ab 1927 dirigierte er im Wiener Rundfunk, wo er
    zwischen 1930 und 1939 noch als Lektor und Zensor arbeitete. Seine Tätigkeit als Dirigent im Inland endete
    allerdings 1934 nach dem Dollfuß-Putsch. Wenige Auslandsengagements führten ihn ab 1929 mehrfach nach London sowie
    nach Deutschland.
</p>

<p i18n>
    Weberns erste kompositorische Zeugnisse sind ab ca. 1900 überliefert. Vor dem Ersten Weltkrieg beschäftigte er sich
    zunächst mit frei-‚atonalen‘ Kompositionen, in den 1920er Jahren entwickelte er dann eine spezifische, radikal
    konzentrierte Form der Zwölftontechnik. Sein Œuvre umfasst von ihm selbst so benannte 31 Opuswerke aus den Jahren
    1908–1944 (vorwiegend bei der Universal Edition veröffentlicht, ab op. 29: posthum publiziert), darüber hinaus
    Bearbeitungen eigener und fremder Werke sowie weitere Kompositionen (teils posthum publiziert) und Skizzen.
</p>

<p i18n>
    Mit seiner Frau Wilhelmine (Minna), geb. Mörtl, die er im Februar 1911 geheiratet hatte, und den vier Kindern
    Amalie, Maria, Peter und Christine lebte Webern vorwiegend in der Umgebung Wiens, zunächst in Mödling, später in
    Maria Enzersdorf. Dort verbrachte das Ehepaar Webern auch die Jahre des Zweiten Weltkriegs. Anfang April 1945
    flüchteten sie nach Mittersill im Oberpinzgau, wo Anton Webern am 15. September von einem amerikanischen
    Besatzungssoldaten erschossen wurde.
</p>
